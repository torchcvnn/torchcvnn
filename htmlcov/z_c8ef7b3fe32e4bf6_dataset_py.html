34def gather_ATRNetSTAR_datafiles( 

35 rootdir: pathlib.Path, 

36 split: Literal["train", "test", "all"], 

37) -> list[str]: 

38 """ 

39 This function gathers all the ATRNet-STAR datafiles from the specified split in the root directory 

40 

41 It returns a list of the paths of the samples (without the extension). 

42 """ 

43 

44 data_files = [] 

45 

46 if split != "all": 

47 split_dir = rootdir / split 

48 else: 

49 split_dir = rootdir 

50 logging.debug(f"Looking for all samples in {split_dir}") 

51 

52 # only look for data files (avoid duplicates) 

53 for filename in split_dir.glob("**/*.mat"): 

54 if not filename.is_file(): 

55 continue 

56 

57 # strip file of the .xml or .mat extension 

58 sample_name = str(filename.with_suffix("")) 

59 

60 # add sample name to the list of known samples 

61 data_files.append(sample_name) 

62 

63 return data_files

66class ATRNetSTARSample: 

67 """ 

68 This class implements a sample from the ATRNet-STAR dataset. 

69 Only slant-range quad polarization complex images are supported. 

70 

71 The extracted complex image is stored in the `data` attribute. 

72 

73 The data is a dictionnary, keys being the polarization ('HH', 'HV', 'VH' or 'VV') 

74 and values a numpy array of the corresponding complex image. 

75 

76 The image annotations are stored in the `annotation` attribute. It contains all the fields 

77 of the XML annotation, in a dictionnary. 

78 

79 Arguments: 

80 sample_name (str): the name of the file to load WITHOUT the .mat or .xml extension 

81 transform (Callable): (optional) A transform to apply to the data 

82 """ 

83 

84 DATA_FILETYPE = ".mat" 

85 HEADER_FILETYPE = ".xml" 

86 

87 POL_KEY_TRANSLATION_DICT = { 

88 "HH": "data_hh", 

89 "HV": "data_hv", 

90 "VH": "data_vh", 

91 "VV": "data_vv", 

92 } 

93 

94 def __init__(self, sample_name: str, transform: Optional[Callable] = None): 

95 self._annotation = xml_to_dict(sample_name + self.HEADER_FILETYPE) 

96 self._data = {} 

97 self.transform = transform 

98 

99 image = scipy.io.loadmat(sample_name + self.DATA_FILETYPE) 

100 for pol in self.POL_KEY_TRANSLATION_DICT.keys(): 

101 self._data[pol] = image[self.POL_KEY_TRANSLATION_DICT[pol]] 

102 

103 @property 

104 def data(self): 

105 if self.transform is not None: 

106 return self.transform(self._data) 

107 return self._data 

108 

109 @property 

110 def annotation(self): 

111 return self._annotation

114class ATRNetSTAR(Dataset): 

115 """ 

116 Implements a PyTorch Dataset for the ATRNet-STAR dataset presented in : 

117 

118 Yongxiang Liu, Weĳie Li, Li Liu, Jie Zhou, Bowen Peng, Yafei Song, Xuying Xiong, Wei Yang, 

119 Tianpeng Liu, Zhen Liu, & Xiang Li. (2025). 

120 ATRNet-STAR: A Large Dataset and Benchmark Towards Remote Sensing Object Recognition in the Wild. 

121 

122 Only slant-range quad polarization complex images are supported. 

123 

124 The dataset is composed of pre-defined benchmarks (see paper). 

125 Dowloading them automatically is possible, but a Hugging Face authentification token 

126 is needed as being logged in is required for this dataset. 

127 

128 Warning : samples are ordered by type, shuffling them is recommended. 

129 

130 Arguments: 

131 root_dir (str): The root directory in which the different benchmarks are placed. 

132 Will be created if it does not exist. 

133 benchmark (str): (optional) Chosen benchmark. If not specified, SOC_40 (entire dataset) will be used as default. 

134 split (str): (optional) Chosen split ('train', 'test' or 'all' for both). Those are pre-defined by the dataset. Default: 'all' 

135 download (bool): (optional) Whether or not to download the dataset if it is not found. Default: False 

136 class_level (str): (optional) The level of precision chosen for the class attributed to a sample. 

137 Either 'category', 'class' or 'type'. Default: 'type' (the finest granularity) 

138 get_annotations (bool): (optional) If `False`, a dataset item will be a tuple (`sample`, `target class`) (default).

140 dictionnary will also be returned: (`sample`, `target class`, `annotation dict`).

239 def __init__( 

240 self, 

241 root_dir: str, 

242 benchmark: Optional[str] = None, 

243 split: Literal["train", "test", "all"] = "all", 

244 download: bool = False, 

245 class_level: Literal["type", "class", "category"] = "type", 

246 get_annotations: bool = False, 

247 transform: Optional[Callable] = None, 

248 ): 

249 super().__init__() 

250 self.root_dir = pathlib.Path(root_dir) 

251 self.split = split 

252 self.class_level = class_level 

253 self.download = download 

254 self.get_annotations = get_annotations 

255 self.transform = transform 

256 

257 if benchmark is None: 

258 # if no benchmark is given, default behavior should be to use the entire dataset (SOC_40) 

259 logging.info( 

260 "No benchmark was specified. SOC_40 (full dataset) will be used." 

261 ) 

262 benchmark = "SOC_40classes" 

263 

264 self.benchmark = benchmark 

265 self._verify_inputs() 

266 

267 if self.benchmark == "SOC_40": 

268 # allow use of the name given to the benchmark in the paper instead of the file 

269 # name actually used in their repository 

270 # (more consistent with the rest of their file naming) 

271 self.benchmark = "SOC_40classes" 

272 

273 self.benchmark_path = self.root_dir / self.benchmark 

274 

275 if not self.benchmark_path.exists(): 

276 if not download: 

277 raise RuntimeError( 

278 f"{self.benchmark} benchmark not found. You can use download=True to download it" 

279 ) 

280 else: 

281 self._download_dataset() 

282 

283 # gather samples 

284 self.datafiles = gather_ATRNetSTAR_datafiles( 

285 rootdir=self.benchmark_path, split=self.split 

286 ) 

287 logging.debug(f"Found {len(self.datafiles)} samples.")

289 def _verify_inputs(self) -> None: 

290 """Verify inputs are valid""" 

291 if self.class_level not in ["type", "class", "category"]: 

292 raise ValueError( 

293 f"Unexpected class_level value. Got {self.class_level} instead of 'type', 'class' or 'category'." 

294 ) 

295 

296 if self.benchmark not in self._ALLOWED_BENCHMARKS: 

297 benchmarks_with_quotes = [f"'{b}'" for b in self._ALLOWED_BENCHMARKS] 

298 raise ValueError( 

299 f"Unknown benchmark. Should be one of {', '.join(benchmarks_with_quotes)} or None" 

300 ) 

301 

302 if self.split not in ["train", "test", "all"]: 

303 raise ValueError( 

304 f"Unexpected split value. Got {self.split} instead of 'train', 'test' or 'all'." 

305 )

307 def _download_dataset(self) -> None: 

308 """ 

309 Downloads the specified benchmark. 

310 Will be placed in a directory named like the benchmark, in root_dir 

311 """ 

312 from .download import check_7z, download_benchmark 

313 

314 check_7z() 

315 download_benchmark( 

316 benchmark=self.benchmark, 

317 root_dir=self.root_dir, 

318 hf_repo_id=self.HF_REPO_ID, 

319 hf_benchmark_path=self.HF_BENCHMARKS_DIR_PATH, 

320 )

322 @property 

323 def classes(self) -> list[str]: 

324 """ 

325 Get the names of all classes at class_level, 

326 """ 

327 match self.class_level: 

328 case "category": 

329 return self.CATEGORIES 

330 

331 case "class": 

332 return self.CLASSES 

333 

334 case "type": 

335 return self.TYPES 

336 

337 case _: 

338 raise ValueError( 

339 f"Unexpected class_level value. Got {self.class_level} instead of type, class or category." 

340 ) 

341 

342 def __len__(self) -> int: 

343 return len(self.datafiles)

345 def __getitem__(self, index: int) -> tuple: 

346 """ 

347 Returns the sample at the given index. Applies the transform 

348 if provided. If `get_annotations` is True, also return the entire annotation dict. 

349 The return type depends on the transform applied and whether or not get_annotations is True 

350 

351 Arguments: 

352 index : index of the sample to return 

353 

354 Returns: 

355 data : the sample 

356 class_idx : the index of the class in the classes list 

357 (Optional) annotation : the annotation dict of the sample. Only if `get_annotations` is True. 

358 

359 """ 

360 sample_name = self.datafiles[index] 

361 sample = ATRNetSTARSample(sample_name=sample_name, transform=self.transform) 

362 class_idx = self.classes.index(sample.annotation["object"][self.class_level]) 

363 

364 if self.get_annotations: 

365 return ( 

366 sample.data, 

367 class_idx, 

368 sample.annotation, 

369 ) 

370 

371 return sample.data, class_idx
